import random

d = {'+':'addition', 
     '-':'soustraction',
     '*':'multiplication',
     '%':'modulo'}

for k, v in d.items():
    print(k, v)
    
score = 0
answer = 'x'
while answer != '':
    x = random.choice(list(d))
    answer = input(d[x]+': ')
    if answer == x:
        score += 1
  
print('score:', score)
